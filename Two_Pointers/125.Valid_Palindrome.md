```java
class Solution {
    public boolean isPalindrome(String s) {
        // Time Complexity: O(n)
        // Space Complexity: O(n)

        // Step 1: Clean the string
        s = s.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();

        // Two-pointer check
        int left = 0;
        int right = s.length() - 1;

        while (left < right) {
            if (s.charAt(left) == s.charAt(right)) {
                left++;
                right--;
            } else {
                return false;
            }
        }

        return true;
    }
}

```
